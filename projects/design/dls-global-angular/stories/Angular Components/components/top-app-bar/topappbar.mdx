# Top App Bar

### API references
The `TopAppBarComponent` is a custom component designed to serve as a top navigation bar for applications.
It can include a logo, application name, and optional hamburger menu functionality.
This component is built using Angular and Angular Material primitives.

### Setting Global Styles and Fonts
[Set Global Styles and Fonts](?path=/docs/installation-setting-global-styles-and-font--overview)


## Usage
Selector: `<ba-top-app-bar>`

### Basic Usage

```typeScript
import { Component } from '@angular/core';
import { TopAppBarComponent } from './top-app-bar.component';
import { IconRegistryModule } from '@dasdigitalplatform/dls-global-angular/icon-registry';
import { TopAppBarComponent } from '@dasdigitalplatform/dls-global-angular/top-app-bar';
import { UserProfileComponent } from '@dasdigitalplatform/dls-global-angular/user-profile';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
    styleUrls: ['./app.component.scss'],
    standalone: true,
    imports: [TopAppBarComponent]
})
export class AppComponent {
  appName = 'My Application'; // The name of the application displayed in the top app bar.

  appLogo = 'my-logo'; // The logo of the application, referenced by its filename.

  homeRoute = '/'; // The route to navigate to when the logo is clicked.

  topAppBarMenu = [
    { label: 'Home', path: '/' }, // Navigation link for the Home page.
    { label: 'About', path: '/about' }, // Navigation link for the About page.
    { label: 'Contact', path: '/contact' } // Navigation link for the Contact page.
  ];
}
```

```html
<ba-top-app-bar
  [appName]="appName"
  [appLogo]="appLogo"
  [homeRoute]="homeRoute"
  [topAppBarMenu]="topAppBarMenu">
</ba-top-app-bar>
```

### Usage with Hamburger Menu Click

```typeScript
import { Component } from '@angular/core';
import { IconRegistryModule } from '@dasdigitalplatform/dls-global-angular/icon-registry';
import { TopAppBarComponent } from '@dasdigitalplatform/dls-global-angular/top-app-bar';
import { UserProfileComponent } from '@dasdigitalplatform/dls-global-angular/user-profile';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
    styleUrls: ['./app.component.scss'],
    standalone: true,
    imports: [TopAppBarComponent]
})
export class AppComponent {
  appName = 'My Application'; // The name of the application displayed in the top app bar.

  appLogo = 'my-logo'; // The logo of the application, referenced by its filename.

  homeRoute = '/'; // The route to navigate to when the logo is clicked.

  topAppBarMenu = [
    { label: 'Home', path: '/' }, // Navigation link for the Home page.
    { label: 'About', path: '/about' }, // Navigation link for the About page.
    { label: 'Contact', path: '/contact' } // Navigation link for the Contact page.
  ];


  /**
   * This method is called when the hamburger menu button is clicked.
   * It is optional for consumers to implement this method.
   * If implemented, it can be used to handle the action that should occur
   * when the hamburger menu is activated, such as opening a side navigation drawer.
   */
  onHamburgerMenuClick() {
    console.log('Hamburger menu clicked!'); // Log message for demonstration purposes.
  }
}
```

```html
<ba-top-app-bar
  [appName]="appName"
  [appLogo]="appLogo"
  [homeRoute]="homeRoute"
  [topAppBarMenu]="topAppBarMenu"
  hamburgerMenuOnly
  (hamburgerMenuClick)="onHamburgerMenuClick()">
</ba-top-app-bar>
```

### Properties

<Markdown>
{`
| Name                                                        | Description                                                                   |
| ------------------------------------------------------------| ----------------------------------------------------------------------------- |
| @Input() menuAlignCenter | boolean | Aligns the menu items to the center if true. Default is true. |
| @Input() hamburgerMenuOnly | boolean | Displays the hamburger menu at all sizes if true. Default is false. |
| @Input() appName | string | The name of the application. Default is 'App Name'. |
| @Input() appLogo | string | A string reference to the application logo. If not passed, no logo displays. |
| @Input() homeRoute | string | Application home route added as a routerLink when clicking the logo. Default is '/'. |
| @Input() isNavigationDrawer | boolean | Checks if the Top App Bar is used with a navigation drawer. Default is false. Consumers must set this to true when using the Top App Bar in conjunction with a navigation drawer to ensure proper UI alignment; otherwise, the UI may be affected. |
| @Input() topAppBarMenu | AtmosphereTopBarNavigationLinkArray | An array of navigation links for the top app bar. |
| @Output() hamburgerMenuClick | EventEmitter<void> | Emits an event when the hamburger menu button is clicked. Consumers can choose to handle this event or not. |
`}
</Markdown>
